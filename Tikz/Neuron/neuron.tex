\newcommand\DENDRITE[4]{ % name, start, stop, fluctuation
    % Compute points in path from start to stop.
    \coordinate (#1_start) at #2;
    \coordinate (#1_stop) at #3;

    \xdef\PtsList{(#1_start)}
    \foreach \i in {1,2,...,10}
    {
        \xdef\PtsList{\PtsList .. (\pts)}
    }
    \path[use Hobby shortcut] (#1) .. \PtsList .. (#3);
}
